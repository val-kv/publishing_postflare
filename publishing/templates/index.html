Платформа для публикации записей

Оплатить подписку var stripe = Stripe('YOUR_PUBLIC_KEY'); var checkoutButton = document.getElementById('checkout-button'); checkoutButton.addEventListener('click', function () { fetch('/create-checkout-session/', { method: 'POST', }).then(function (response) { return response.json(); }).then(function (sessionId) { return stripe.redirectToCheckout({ sessionId: sessionId.id }); }).then(function (result) { if (result.error) { alert(result.error.message); } }); });
